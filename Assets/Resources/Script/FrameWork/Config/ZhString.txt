local localizationManager = LocalizationManager.Instance
function localizeComplexString(value)
    for variant, text in pairs(value) do
        local localizedText = localizationManager:GetLocalizedText(text)
        if localizedText then
            localizedText = string.gsub(localizedText, "'(.-)'", "\"%1\"")
            value[variant] = localizedText
        end
    end
end

function localizeZhString()
    for key, value in pairs(ZhString) do
        if type(value) == "string" then
            local localizedText = localizationManager:GetLocalizedText(value)
            if localizedText then
                localizedText = string.gsub(localizedText, "'(.-)'", "\"%1\"")
                ZhString[key] = localizedText
            end
        elseif type(value) == "table" then
            -- ถ้าค่าเป็นตารางและมีโครงสร้างซับซ้อน เช่น SkillTip_DelayCDTime
            localizeComplexString(value)
        end
    end
end
